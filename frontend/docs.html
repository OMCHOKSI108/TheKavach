Authentication

All endpoints except /generate-key, /health, and /status require an API key.

Authorization: Bearer tk_your_api_key_here

POST /api/generate-key

Generate a new API key.

curl -X POST http://localhost:8000/api/generate-key \
  -H "Content-Type: application/json" \
  -d '{"name": "John Doe"}'
import requests

response = requests.post(
    "http://localhost:8000/api/generate-key",
    json={"name": "John Doe"}
)
data = response.json()
api_key = data["api_key"]
print(f"Your API key: {api_key}")
const response = await fetch("http://localhost:8000/api/generate-key", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ name: "John Doe" })
});
const data = await response.json();
console.log("Your API key:", data.api_key);
#include <iostream>
#include <curl/curl.h>

size_t WriteCallback(void* contents, size_t size, size_t nmemb, std::string* output) {
    output->append((char*)contents, size * nmemb);
    return size * nmemb;
}

int main() {
    CURL* curl = curl_easy_init();
    std::string response;
    const char* json = R"({"name": "John Doe"})";

    curl_easy_setopt(curl, CURLOPT_URL, "http://localhost:8000/api/generate-key");
    curl_easy_setopt(curl, CURLOPT_POST, 1L);
    curl_easy_setopt(curl, CURLOPT_POSTFIELDS, json);
    curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, WriteCallback);
    curl_easy_setopt(curl, CURLOPT_WRITEDATA, &response);

    struct curl_slist* headers = NULL;
    headers = curl_slist_append(headers, "Content-Type: application/json");
    curl_easy_setopt(curl, CURLOPT_HTTPHEADER, headers);

    CURLcode res = curl_easy_perform(curl);
    if (res == CURLE_OK) std::cout << response << std::endl;

    curl_easy_cleanup(curl);
    curl_slist_free_all(headers);
    return 0;
}

GET /api/logs

Fetch synthetic cybersecurity logs. Supports filtering.

# Fetch 100 logs
curl "http://localhost:8000/api/logs?count=100" \
  -H "Authorization: Bearer tk_your_key"

# Filter by threat label
curl "http://localhost:8000/api/logs?count=50&threat_label=malicious" \
  -H "Authorization: Bearer tk_your_key"

# Filter by protocol
curl "http://localhost:8000/api/logs?count=50&protocol=TCP" \
  -H "Authorization: Bearer tk_your_key"
import requests

API_KEY = "tk_your_key"

# Fetch logs

    "http://localhost:8000/api/logs",
    headers={"Authorization": f"Bearer {API_KEY}"},
    params={"count": 100, "threat_label": "malicious"}

logs = response.json()
print(f"Received {logs['count']} logs")

for log in logs["logs"]:
    print(f"[{log['threat_label']}] {log['source_ip']} -> {log['dest_ip']} ({log['protocol']})")

GET /api/stream

Server-Sent Events (SSE) for real-time log streaming.

# Stream logs at 0.5s interval
curl -N "http://localhost:8000/api/stream?interval=0.5" \
  -H "Authorization: Bearer tk_your_key"

# Stream only malicious logs
curl -N "http://localhost:8000/api/stream?interval=1&threat_label=malicious" \
  -H "Authorization: Bearer tk_your_key"
import requests
import json

API_KEY = "tk_your_key"

response = requests.get(
    "http://localhost:8000/api/stream?interval=0.5",
    headers={"Authorization": f"Bearer {API_KEY}"},
    stream=True
)

for line in response.iter_lines():
    if line:
        line = line.decode("utf-8")
        if line.startswith("data: "):
            log = json.loads(line[6:])
            print(f"[{log['threat_label']}] {log['source_ip']} -> {log['dest_ip']}")
const EventSource = require("eventsource");

const evtSource = new EventSource(
    "http://localhost:8000/api/stream?interval=0.5",
    { headers: { "Authorization": "Bearer tk_your_key" } }
);

evtSource.onmessage = (event) => {
    const log = JSON.parse(event.data);
    console.log([]  -> );
};

evtSource.onerror = (err) => {
    console.error("Stream error:", err);
    evtSource.close();
};

Query Parameters

ParameterEndpointTypeDefaultDescription

threat_label/logs, /streamstring-benign, suspicious, malicious
log_type/logsstring-firewall, network, application, ids
protocol/logsstring-TCP, UDP, HTTP, HTTPS, etc.

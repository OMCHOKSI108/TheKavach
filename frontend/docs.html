API Documentation

Base URL

https://your-app.onrender.com/api

Local: http://localhost:8000/api

Authentication

All endpoints except /generate-key, /health, and /status require an API key.

Authorization: Bearer tk_your_api_key_here

POST /api/generate-key

curl -X POST http://localhost:8000/api/generate-key \
  -H "Content-Type: application/json" \
  -d '{"name": "John Doe"}'
import requests
response = requests.post(
    "http://localhost:8000/api/generate-key",
    json={"name": "John Doe"}
)
data = response.json()
print(f"API key: {data['api_key']}")
const res = await fetch("http://localhost:8000/api/generate-key", {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ name: "John Doe" })
});
const data = await res.json();
console.log("API key:", data.api_key);
#include <iostream>
#include <curl/curl.h>

size_t WriteCallback(void* contents, size_t size, size_t nmemb, std::string* out) {
    out->append((char*)contents, size * nmemb);
    return size * nmemb;
}

int main() {
    CURL* curl = curl_easy_init();
    std::string response;
    curl_easy_setopt(curl, CURLOPT_URL, "http://localhost:8000/api/generate-key");
    curl_easy_setopt(curl, CURLOPT_POSTFIELDS, R"({"name":"John Doe"})");
    curl_easy_setopt(curl, CURLOPT_WRITEFUNCTION, WriteCallback);
    curl_easy_setopt(curl, CURLOPT_WRITEDATA, &response);
    struct curl_slist* headers = NULL;
    headers = curl_slist_append(headers, "Content-Type: application/json");
    curl_easy_setopt(curl, CURLOPT_HTTPHEADER, headers);
    curl_easy_perform(curl);
    std::cout << response << std::endl;
    curl_easy_cleanup(curl);
    return 0;
}

log_type/logsstring-firewall, network, application, ids

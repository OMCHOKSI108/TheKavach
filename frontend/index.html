Quick Start

# Generate API key
curl -X POST http://localhost:8000/api/generate-key \
  -H "Content-Type: application/json" \
  -d '{"name": "Your Name"}'

# Fetch logs
curl http://localhost:8000/api/logs?count=100 \
  -H "Authorization: Bearer tk_your_key"

# Stream logs
curl http://localhost:8000/api/stream?interval=0.5 \
  -H "Authorization: Bearer tk_your_key"

# Python example
import requests
response = requests.get("http://localhost:8000/api/logs",
    headers={"Authorization": "Bearer tk_your_key"},
    params={"count": 100, "threat_label": "malicious"})
logs = response.json()

Quick Start

# 1. Generate API key
curl -X POST http://localhost:8000/api/generate-key \
  -H "Content-Type: application/json" \
  -d '{"name": "Your Name"}'

# 2. Fetch logs
curl http://localhost:8000/api/logs?count=100 \
  -H "Authorization: Bearer tk_your_key"

# 3. AI analyze a log
curl -X POST http://localhost:8000/api/ai/analyze \
  -H "Content-Type: application/json" \
  -d '{"protocol":"TCP","action":"blocked","user_agent":"Nmap Scripting Engine","request_path":"/admin/config","bytes_transferred":45000,"log_type":"firewall"}'

# 4. Stream logs
curl http://localhost:8000/api/stream?interval=0.5 \
  -H "Authorization: Bearer tk_your_key"

# 5. Python AI analysis
import requests
response = requests.post("http://localhost:8000/api/ai/analyze",
    json={"protocol":"TCP","action":"blocked","user_agent":"Nmap",
          "request_path":"/admin/config","bytes_transferred":45000,"log_type":"firewall"})
result = response.json()
print(f"Threat: {result['threat']}, Severity: {result['severity']}")
